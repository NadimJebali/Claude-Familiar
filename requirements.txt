# Runtime dependencies for the Claude Familiar widget.
#
# As of ADR-0001 (docs/adr/0001-runtime-dependencies.md) the project permits
# required third-party packages here, with OS-specific ones gated by environment
# markers (e.g. `pywin32; sys_platform == "win32"`).
#
# Dev/test tools (pytest, hypothesis, ruff) live in requirements-dev.txt.

# Qt for Python — the widget's view layer (ADR-0002 / PRD #51). The migration is a
# hard cutover to PySide6; once it lands it replaces the Tk view and subsumes the
# tray + toast deps below (pystray/Pillow/plyer are removed at the cutover, #63).
PySide6>=6.8

# Cross-platform system tray (#18). pystray renders the tray icon + menu on
# Windows / Linux / macOS; Pillow supplies the icon image.
pystray>=0.19
Pillow>=10

# Native OS toast notifications (#19) for permission/attention + usage-limit
# events, complementing the in-app speech bubble.
plyer>=2.1

# Cross-platform process detection / liveness (#17): finding the owning Claude
# process (hooks) and pruning a session when it dies (widget). Imported lazily and
# best-effort — if absent, owner tracking degrades to the staleness backstop.
psutil>=5.9

# Windows-only (#20): robust Start-menu / desktop / run-at-login .lnk creation via
# WScript.Shell COM, replacing a fragile PowerShell subprocess. Lazy-imported only
# on Windows; the Linux .desktop path needs nothing extra.
pywin32>=306; sys_platform == "win32"
